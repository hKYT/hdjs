            position: absolute;
            width: 100%;

    <script>
        $(document).ready(function() {
            var myCar = document.getElementById("car");  
            var icar = document.getElementById("icar");  
            var sf = new HDjs.ScreenFit();
            setCar()
            $(window).resize(function() {    
                setCar()        
            })
            function setCar(){            
                icar.style.width=sf.scale()+"%";
                myCar.style.left=sf.getX(843)+"px";
                myCar.style.top=sf.getY(724)+"px";

            }
        })
    </script>
