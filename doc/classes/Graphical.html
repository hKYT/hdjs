        var cxt = document.getElementById("myCanvas");
        cxt.width = window.innerWidth;
        cxt.height = window.innerHeight;
        var con = cxt.getContext('2d');
        con.fillRect(0, 0, cxt.width,cxt.height);

        var mygraphical=new HDjs.Graphical();
        mygraphical.createLine(con,{x:230,y:300},{x:50,y:50},3,"rgba(23,255,255,1)")
        mygraphical.createPoint(con,{x:230,y:100},50,"rgba(23,24,255,1)",1)
        mygraphical.createCurve(con,{x:430,y:10},{w:200,h:100},"rgba(255,234,25,1)",5,2)
    </script>
